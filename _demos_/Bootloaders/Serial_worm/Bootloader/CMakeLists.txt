add_compile_options(-Wall
        -Wno-format          # int != int32_t as far as the compiler is concerned because gcc has int32_t as long int
        -Wno-unused-function # we have some for the docs that aren't called
        -Wno-maybe-uninitialized
        -Os)

add_executable(Worm_Bootloader)

target_sources(Worm_Bootloader PRIVATE serial_worm.c)

pico_set_linker_script(Worm_Bootloader worm_bootloader.ld)

# Pull in our pico_stdlib which pulls in commonly used features
target_link_libraries(Worm_Bootloader pico_stdlib hardware_dma hardware_uart hardware_flash hardware_watchdog hardware_pwm)

# create map/bin/hex file etc.
pico_add_extra_outputs(Worm_Bootloader)
